## Callout

<Callout type="note" title="Note">
  Token validation runs before the proxy forwards any request body.
</Callout>

<Callout type="tip" title="Tip">
  Use `make dev-smoke` after changing proxy middleware order.
</Callout>

<Callout type="warning" title="Warning">
  Rate limiting fails open when Redis is unavailable.
</Callout>

<Callout type="danger" title="Danger">
  Never commit PATs or API keys to the repository.
</Callout>

## Steps

<Steps>
  <Step title="Start the stack">
    Run `make compose-dev-up` and wait for Postgres on port 5432.
  </Step>
  <Step title="Apply migrations">
    Run `make db-migrate` to create core tables and RLS policies.
  </Step>
  <Step title="Issue a token">
    Use the auth service gRPC API to create an org-scoped PAT.
  </Step>
</Steps>

```ts title="src/middleware.ts"
import { rateLimit } from "./config";

export function middleware(req: Request) { // [!code highlight]
  const id = req.headers.get("x-org-id");
  if (!id) return new Response("Unauthorized", { status: 401 });

  const allowed = rateLimit.check(id); // [!code highlight]
  if (!allowed) return new Response("Too Many Requests", { status: 429 }); // [!code highlight]

  return next(req);
}
```
